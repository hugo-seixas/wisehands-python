// dado um numero secreto
numero_secreto = random(1,50)

// lendo valor de entrada pelo usuário
entrada = input ("digite um número inteiro entre 1 e 50)

// verificação do número
if (entrada == numero_secreto)
   print ("Voce acertou")

else if (entrada > numero_secreto)
   print ("Digite um número menor")

else
    print ("Digite um número maior")
